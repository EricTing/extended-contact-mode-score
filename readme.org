    import pandas as pd

    def QuantileCoef(dat):
        """
        Keyword Arguments:
        dat -- dis(similarity) values
        """
        q1, q3 = np.percentile(dat, [25, 75])
        return (q3 - q1) / (q3 + q1)


    cms_qe = {}
